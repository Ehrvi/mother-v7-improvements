### Implementation Plan

**Step 1: Create Component**
```typescript
// client/src/components/HealthCheck.tsx
import { trpc } from '@/lib/trpc';
import { Card } from '@/components/ui/card';
import { Badge } from '@/components/ui/badge';

export function HealthCheck() {
  const { data: health, isLoading } = trpc.system.health.useQuery(undefined, {
    refetchInterval: 60000, // 60s
  });
  
  if (isLoading) return <div>Loading...</div>;
  
  return (
    <div className="grid gap-4">
      <Card>
        <h2>System Status</h2>
        <Badge variant={health.online ? 'success' : 'destructive'}>
          {health.online ? '✅ Online' : '❌ Offline'}
        </Badge>
        <p>Response Time: {health.responseTime}ms</p>
      </Card>
      
      <Card>
        <h2>Feature Validation</h2>
        {health.features.map(f => (
          <div key={f.name}>
            {f.status ? '✅' : '❌'} {f.name}
          </div>
        ))}
      </Card>
      
      <Card>
        <h2>Metrics</h2>
        <p>Total Queries: {health.metrics.totalQueries}</p>
        <p>Avg Quality: {health.metrics.avgQuality}/100</p>
        <p>Avg Response Time: {health.metrics.avgResponseTime}s</p>
        <p>Knowledge Base: {health.metrics.knowledgeBaseSize} entries</p>
        <p>Learned Entries: {health.metrics.learnedEntries}</p>
      </Card>
    </div>
  );
}
```

**Step 2: Add Backend Endpoint**
```typescript
// server/routers/system.ts
health: publicProcedure.query(async () => {
  const startTime = Date.now();
  
  // Test query
  const testResult = await processQuery("test", undefined);
  const responseTime = Date.now() - startTime;
  
  // Get metrics
  const stats = await getQueryStats();
  const kbSize = await db.select({ count: sql`count(*)` }).from(knowledge);
  const learnedCount = await db.select({ count: sql`count(*)` })
    .from(knowledge)
    .where(eq(knowledge.category, 'learned'));
  
  return {
    online: true,
    responseTime,
    features: [
      { name: 'Multi-Tier Routing', status: testResult.tier !== null },
      { name: 'Quality Scoring', status: testResult.quality.qualityScore > 0 },
      { name: 'Knowledge Base', status: kbSize[0].count > 0 },
      { name: 'Continuous Learning', status: learnedCount[0].count > 0 },
      { name: 'Creator Context', status: true }, // Always true if deployed
      { name: 'ReAct Pattern', status: testResult.reactObservations !== null },
      { name: 'Vector Search', status: true }, // Check embeddings exist
      { name: 'Phase 2 Quality', status: testResult.quality.coherenceScore !== null },
    ],
    metrics: {
      totalQueries: stats.totalQueries,
      avgQuality: stats.avgQuality,
      avgResponseTime: stats.avgResponseTime,
      knowledgeBaseSize: kbSize[0].count,
      learnedEntries: learnedCount[0].count,
    },
  };
}),
```
